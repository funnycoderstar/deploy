# 插入排序

// 插入排序
this.insertionSort = function() {
    let len = array.length;
    let j;
    let temp;
    for(let i = 0; i < length; i++) {
        j = i;
        temp = array[i];
        while( j > 0 && array[j - 1] > temp) {
            array[j] = array[j - 1];
            j--;
        }
        array[j] = temp;
    }
}
